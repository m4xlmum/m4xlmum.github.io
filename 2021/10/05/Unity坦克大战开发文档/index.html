碰撞检测

2D的碰撞器为，创建脚本的那里选择Box collider 2D。

对预制体(prefabs)进行操作的时候，老师讲的是有个apply的一个操作的，但是注意到unity 2020中并没有这个apply的操作(我后来找到了，在这里，如下图)，似乎是直接改变预制体的属性，Hierarchy中的属性就会也跟着更改。

令，对移动的物体创建刚体(Rigidbody 2D)的属性，这里我们只对坦克创建刚体的属性，其他map相关的地方，除了草丛不用创建碰撞器，其他的都需要创建碰撞器。

坦克的攻击方法

需要编辑Player的脚本，先打开脚本

目标：玩家点击空格键时实例化出来一颗子弹

首先拿到子弹预制体的引用，然后定义按下空格键时的动作就行了

public GameObject BulletPrefab;  //拿到子弹的预制体的引用

/**
     坦克的攻击方法
         */
private void Attack()
{
    if (Input.GetKeyDown(KeyCode.Space))
    {
        Instantiate(BulletPrefab, transform.position, transform.rotation);
        //第一个参数是要生成的预制体，第二个是生成预制体时的位置，第三个时生成预制体的旋转
    }
}

这之后再将Attack函数添加到update函数里面（这里如果将该函数放在fixedupdate里面会产生按空格发不出来子弹的Bug），再将BulletPrefab的预制体赋一下值。

子弹方向的问题解决

因为Unity2D是左手坐标系，也就是说Z轴是从前到后的，所以说向左旋转是Vector3(0, 0, -90)。向右则相反。

上下则很简明，向下就是(±180°)都可以。

public Vector3 BulletEulerAngles;  //定义子弹的欧拉角

// 子弹的角度应为：当前坦克的角度 + 子弹应该旋转的角度
Instantiate(BulletPrefab, transform.position, Quaternion.Euler(transform.eulerAngles + BulletEulerAngles));

把更改子弹的旋转角的位置放到坦克移动那里，也就是说，坦克朝哪个方向移动，子弹就往哪个方向射出。此处举出一例

if (x > 0) //向右
{
    spriteRenderer.sprite = tankSprite[1];
    BulletEulerAngles = new Vector3(0, 0, 90);
}
else if (x < 0) //向左
{
    spriteRenderer.sprite = tankSprite[3];
    BulletEulerAngles = new Vector3(0, 0, -90); //Z轴旋转-90°
}

这里的正负90度也是涉及到了左手坐标系的问题，这里就不再写了。

左手坐标系问题再解决

老师的那里的问题，我这还是标准的右手坐标系啊，所以我这里的旋转的角度还是顺时针旋转的，所以这里我的向右的方向的v3的坐标应该是(0,0,90)

地图的实例化

目标：把整个地图的场景搭建起来。

调地图大小的时候把相机改为正交模式即可更改Size，具体的操作流程如下链接。

我这里的地图的纵向长大概为(16)，横向长大概为(24)。

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class MapCreator : MonoBehaviour
{
    //用来装饰地图初始化所需物体的数组
    //0.老家 1.墙 2.障碍 3.出生效果 4.河流 5.草 6.空气墙
    public GameObject[] Items;


    //已经有东西的位置的列表
    private List<Vector3> itemPositionList = new List<Vector3>();

    //地图实例化
    private void Awake()
    {
        //实例化老家
        CreateItem(Items[0], new Vector3(0, -8, 0), Quaternion.identity);

        //用墙把家围起来
        CreateItem(Items[1], new Vector3(-1, -8, 0), Quaternion.identity);
        for (int i = 0; i < 3; i++)
        {
            CreateItem(Items[1], new Vector3(i-1, -7, 0), Quaternion.identity);
        }
        CreateItem(Items[1], new Vector3(1, -8, 0), Quaternion.identity);

        //实例化外围空气墙,
        //上下
        for (float i = -12; i < 13; i++)
        {
            CreateItem(Items[6], new Vector3(i, 9, 0), Quaternion.identity);
        }
        for (float i = -12; i < 13; i++)
        {
            CreateItem(Items[6], new Vector3(i, -9, 0), Quaternion.identity);
        }
        //左右
        for (float i = -8; i < 9; i++)
        {
            CreateItem(Items[6], new Vector3(13, i, 0), Quaternion.identity);
        }
        for (float i = -8; i < 9; i++)
        {
            CreateItem(Items[6], new Vector3(-13, i, 0), Quaternion.identity);
        }


        //实例化地图
        //1.墙
        for (int i = 0; i < 60; i++)
        {
            CreateItem(Items[1], CreateRandomPosition(), Quaternion.identity);
        }
        //2.障碍
        for (int i = 0; i < 25; i++)
        {
            CreateItem(Items[2], CreateRandomPosition(), Quaternion.identity);
        }
        //4.河流
        for (int i = 0; i < 30; i++)
        {
            CreateItem(Items[4], CreateRandomPosition(), Quaternion.identity);
        }
        //5.草
        for (int i = 0; i < 25; i++)
        {
            CreateItem(Items[5], CreateRandomPosition(), Quaternion.identity);
        }

        //初始化玩家
        GameObject born = Instantiate(Items[3], new Vector3(-2, -8.25f, 0), Quaternion.identity);
        born.GetComponent<Born>().createPlayer = true;


        //初始化敌人
        CreateItem(Items[3], new Vector3(-12, 8, 0), Quaternion.identity);
        CreateItem(Items[3], new Vector3(0, 8, 0), Quaternion.identity);
        CreateItem(Items[3], new Vector3(12, 8, 0), Quaternion.identity);

        InvokeRepeating("createEnemy", 4, 5);
    }


    private void CreateItem(GameObject item, Vector3 itemPosition, Quaternion itemRotation)
    {
        GameObject itemGo = Instantiate(item, itemPosition, itemRotation);
        itemGo.transform.SetParent(gameObject.transform);

        //生成的位置保存一下
        itemPositionList.Add(itemPosition);

        //判断位置是否已经生成过item
    }


    //产生随机位置的方法
    private Vector3 CreateRandomPosition()
    {
        //最外圈的地方不产生物品
        while (true)
        {
            Vector3 createPosition = new Vector3(Random.Range(-11, 12), Random.Range(-7, 8), 0);
            if (!HasItem(createPosition))
            {
                return createPosition;
            }
        }
    }


    //判断位置是否已经实例化物品的方法
    private bool HasItem(Vector3 createPosition)
    {
        foreach (var item in itemPositionList)
        {
            if (createPosition == item)
            {
                return true;
            }
        }
        return false;
    }


    //产生敌人的方法
    private void createEnemy()
    {
        int num = Random.Range(0, 3);
        Vector3 EnemyPosition = new Vector3();
        if (num == 0)
        {
            EnemyPosition = new Vector3(-12, 8, 0);
        }
        else if (num == 1)
        {
            EnemyPosition = new Vector3(0, 8, 0);
        }
        else
        {
            EnemyPosition = new Vector3(12, 8, 0);
        }

        CreateItem(Items[3], EnemyPosition, Quaternion.identity);
    }
}

玩家状态管理

脚本实现对玩家状态的一个管理

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

public class PlayerManager : MonoBehaviour
{
    //玩家的生命
    public int playerLife = 3;
    //玩家的得分
    public int playerScore = 0;
    //玩家是否死亡
    public bool isDead;
    //游戏是否失败
    public bool isDefeat = false;
    //出生特效的引用
    public GameObject reLivePrefab;
    //两个ui得引用
    public Text playerScoreText;
    public Text playerLifeText;
    //拿到失败的图片
    public GameObject imgGameOver;

    private static PlayerManager instance;
    public static PlayerManager Instance { get => instance; set => instance = value; }


    private void Awake()
    {
        instance = this;
    }
    // Start is called before the first frame update
    void Start()
    {

    }

    // Update is called once per frame
    void Update()
    {
        //游戏结束
        if (isDefeat)
        {
            imgGameOver.SetActive(true);
            return;
        }

        if (isDead)
        {
            reLive();
        }
        playerLifeText.text = playerScore.ToString();
        playerLifeText.text = playerLife.ToString();
    }

    //复活
    private void reLive()
    {
        if (playerLife == 0)
        {
            //没命了，回到主界面
        }
        else
        {
            GameObject born = Instantiate(reLivePrefab, new Vector3(-2, -8.25f, 0), Quaternion.identity);
            born.GetComponent<Born>().createPlayer = true;
            isDead = false;
        }
    }
}

制作开始界面，第一个场景

直接先创建start scene。

然后直接拉ui，但是这里的选项有个脚本。

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.SceneManagement;

public class Option : MonoBehaviour
{
    private int chose = 0;
    public Transform onePlayer;
    public Transform twoPlayer;

    // Start is called before the first frame update
    void Start()
    {
        
    }

    // Update is called once per frame
    void Update()
    {
        //获得输入
        if (Input.GetKeyDown(KeyCode.W))
        {
            chose = 0;
            transform.position = onePlayer.position;
        }
        if (Input.GetKeyDown(KeyCode.S))
        {
            chose = 1;
            transform.position = twoPlayer.position;
        }

        //更换场景
        if (chose == 0 && Input.GetKeyDown(KeyCode.Space))
        {
            SceneManager.LoadScene("GameScene");
        }
        else if (chose == 1 && Input.GetKeyDown(KeyCode.Space))
        {
            SceneManager.LoadScene("GameScene");
        }
    }
}
