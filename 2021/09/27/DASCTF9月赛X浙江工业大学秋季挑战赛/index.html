def get_ed(p, q):
	k = cal_bit(q*p)
	phi_n = (p-1)*(q-1)
	r = random.randint(10, 99)

2. 加密分析

这个加密脚本其实可以分解为下面几个式子：

而其中我们可以根据中间的那个式子算出t的数值。然后注意到t的值很大，并且它的模数也是phin，那我们就能对第一个式子采用维纳攻击，获得u和phin，这里u并没有什么用处。获取phin之后，就能根据第三个式子，直接求出d的数值了，这时直接c、d、nRSA解密三要素已经有了，直接解密即可。

所以根据上面的分析，这里的难点就是维纳攻击如何实现了。

3. 维纳攻击的实现

pablocelayes/rsa-wiener-attack

维纳攻击的实现脚本主要就是上面的这个，下载解压他这个代码之后，修改RSAwienerHacker.py文件的hack_RSA(e, n)的返回值，加上个phi，因为这里我们需要phin的值而不是u。

然后直接在解密脚本中调用即可。

hint1的代码也很简单，如下

hint1 = (invert(e, lcm(p - 1, q - 1))) % (p - 1)

这里注意到这个lcm(p - 1, q - 1)与phin = (p - 1) * (q - 1)的值是非常接近的，后者是前者的倍数，但是这个倍数不大。invert(e, lcm(p - 1, q - 1))可以看做invert(e, phin)也就是d(其实不是，但是他们mod (p-1)之后就相等了，因为invert(e, phin) - invert(e, lcm(p - 1, q - 1))后的值肯定是(p-1)的倍数，那这个hint1也是dp。

c = m ** e mod n
dp = d mod (p-1)

c**dp = m**(e*dp) mod n
c**dp mod p = m**(e*dp) mod p
e*dp = e*d mod (p-1) = 1 mod (p-1)
c**dp mod p = m**(1+k*(p-1)) mod p

import xlrd
from decimal import Decimal
import re


def transform(amount):
    """
    摘自: https://blog.csdn.net/farewellnec/article/details/103686077
    """
    chinese_num = {'零': 0, '壹': 1, '贰': 2, '叁': 3, '肆': 4, '伍': 5, '陆': 6, '柒': 7, '捌': 8, '玖': 9}
    chinese_amount = {'分': 0.01, '角': 0.1, '元': 1, '拾': 10, '佰': 100, '仟': 1000, '圆': 1}
    amount_float = 0
    amount_yuan = 0
    for i in chinese_amount:
        if i in amount:
            if amount[amount.index(i) - 1] in chinese_num.keys():
                amount_yuan += chinese_num[amount[amount.index(i) - 1]] * chinese_amount[i]
    amount_float += amount_yuan
    return amount_float


def transform2(num):
    chinese_num = {'零': 0, '壹': 1, '贰': 2, '叁': 3, '肆': 4, '伍': 5, '陆': 6, '柒': 7, '捌': 8, '玖': 9}
    return chinese_num[num]

    tmp2 = transform2(data2[i])
