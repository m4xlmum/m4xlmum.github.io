打包遇到问题

​ 这里使用官网推荐的打包方案时出现了一点点小小的错误，如下所示，修了半天也没修好，索性换一种打包方案了。

另一种打包方案

electron-packager打包可执行文件

npm i electron-packager --save-dev
  1. 在package.json中添加下面的script。
"scripts": {
    "package": "electron-packager . ElectronDemo --platform=win32 --arch=x64 --icon=computer.ico --out=./out --app-version=0.0.1 --overwrite --ignore=node_modules"
  },
  1. 打包可执行文件命令
$ npm run package

electron-builder打包安装包

npm i electron-builder --save-dev
  1. 在package.json文件中添加一下代码:
"build": {
    "appId": "com.example.app",
    "mac": {
      "target": [ // 目标包类型
        "dmg",
        "zip"
      ]
    },
    "win": {
      "target": [ // 目标包类型
        "nsis",
        "zip"
      ]
    }
  },
 "scripts": {
    "dist": "electron-builder"
  },
  1. 自定义安装路径，要在buid字段里面添加个nsis的节点，其中有2个配置项是关键点：
  • oneClick设置为false
  • allowToChangeInstallationDirectory设置为true
"build": {
    "appId": "com.example.app",
    "mac": {
      "target": [
        "dmg",
        "zip"
      ],
      "icon": "img.ico"
    },
    "win": {
      "target": [
        "nsis",
        "zip"
      ],
      "icon": "img.ico" // 更换桌面上生成的图标
    },
    "nsis": {
      "oneClick": false, // 是否一键安装，建议为 false，可以让用户点击下一步、下一步、下一步的形式安装程序，如果为true，当用户双击构建好的程序，自动安装程序并打开，即：一键安装
      "allowToChangeInstallationDirectory": true, // 允许修改安装目录，建议为 true，是否允许用户改变安装目录，默认是不允许
      "createDesktopShortcut": true, // 创建桌面图标
      "createStartMenuShortcut": true, // 创建开始菜单图标
      "installerIcon": "test.ico", // 安装图标
      "uninstallerIcon": "xxx.ico", // 卸载图标
      "installerHeaderIcon": "xxx.ico" // 安装时头部图标
    },
    "scripts": {
      "dist": "electron-builder"
    }
  }

上面的json内容仅为示例，运行下面的打包命令的时候要把注释去掉，还有你不想设置icon图标的话，去掉即可，所有图标就是默认的了。

  1. 打包所使用的命令为：
$ npm run dist

上面都配置好后打包命令

​ 把上面的东西都搞好之后，再重新打包就简单多了，直接两个命令即可解决。

npm run package && npm run dist

Electron连接数据库

​ 怎么说呢？这个客户端连接数据库的问题，我想着应该是不能直接将数据库连接方式写到这个程序里面的，这个程序如果被逆向将会造成非常严重的数据泄露，所以最好还是把数据库服务放到服务端里面，然后本地进行查询数据库的时候，只需要访问服务端的接口即可。这里为了磨练自己的node开发能力，这里的服务端的数据库连接接口我决定使用node来编写。

连接Mysql测试

  1. 首先npm init一个server的文件夹，node的入口设为main.js
  2. npm install mysql下载数据库模块
  3. 新建main.js文件，并将以下代码写入文件
const mysql      = require('mysql');
const connection = mysql.createConnection({
  host     : 'localhost',
  user     : 'admin',
  password : '123456',
  database : 'user'
});
 
connection.connect();
 
connection.query('SELECT 1 + 1 AS solution', function (error, results, fields) {
  if (error) throw error;
  console.log('The solution is: ', results[0].solution);
});

Mysql数据查询测试(异步问题解决)

​ 这个查询数据与返回数据到前台的时候有一个node的异步的、非阻塞的流程导致的小问题，就是在接收返回值的时候得到了undefined，研究了半天，才解决了。

const mysql      = require('mysql');


function createconnection() {
    const connection = mysql.createConnection({
        host     : 'localhost',
        user     : 'admin',
        password : '123456',
        database : 'user'
      });
    return connection;
}


function GetPassword(username) {
    const connection = createconnection();
    connection.connect();
    const sql = `SELECT PASSWORD FROM user WHERE username="${username}"`;
    console.log(sql);
    const promise = new Promise((res, rej) => 
        {
            connection.query(sql, function (error, results) {
                if (error) {
                    res(error.message);
                }
                else {
                    //坑1 查询的数据不是json 是一个rowdatapocket对象
                    let dataString = JSON.stringify(results);
                    let data = JSON.parse(dataString);
                    res(data);
                }
                connection.end();
            });
            
        }
    ) 
    return promise;

}

function VertifyUser(username, password) {
    const promise = new Promise((res, rej) => {
        GetPassword(username, password).then((results, reject) => 
        {
            //在这里写验证代码
            let realpass = results[0]["PASSWORD"];
            console.log(password);
            if (password == realpass) {
                res("success");
                console.log("密码正确");
            }
            else {
                res("failed");
                console.log("密码错误");
            }
        });
    });
    return promise;
}

Express做服务端路由解析

​ 好了，接下来加入express模块来做路由。npm install express

/*
 *使用express做路由解析
*/
const express = require('express');
const app = express();

const port = 8000;
app.listen(port, () => {
  console.log('We are live on ' + port);
});

//url解析
app.use(express.urlencoded({ extended: true }));


//登录验证模块
app.post('/login', function(req, res) {
    let body = req.body;
    let username = body["username"];
    let password = body["password"];
    VertifyUser(username, password).then((results, reject) => {
        res.send(results);
    });
});
