def ext_euclid(a, b):
    if b == 0:
        return 1, 0, a
    else:
        x, y, q = ext_euclid(b, a % b)  # q = gcd(a, b) = gcd(b, a%b)
        x, y = y, (x - (a // b) * y)
        return x, y, q


def same_mod(n, e1, e2, c1, c2):
    s, t, q = ext_euclid(e1, e2)
    m = (gmpy2.powmod(c1, s, n) * gmpy2.powmod(c2, t, n)) % n  # powmod进行大数运算
    return m


if __name__ == '__main__':

CBC字节翻转攻击(Cryptography)

​ 专门在后面加了一个密码学的标签是为了到时候防止跟web的字节翻转攻击互相混淆了。这个题目极具代表性的特征是有个register的功能与

for i in range(3):的三次选择(一次用来加密、一次用来翻转第二组明文、一次用来翻转第一组明文<通过修改初始向量iv的方式>)

​ 先放一个cbc模式的图片方便后续的讲解：

​ 一般情况下(常见套路下)的CBC字节翻转攻击面对的情景都是只有两块内容的块加密，此时一般会让更改第二块明文的内容，也就是下图中红圈所圈住的内容是需要被更改的

​ 而这个密文是可以被更改的，更改所需要的条件为我们可以知道前一组的密文、所需要更改的内容更改前的明文、所需要更改的内容更改后的明文而将他们在一起异或之后获得的内容作为前一组的密文即可。原理为：

如果：A1=更改前第一组密文, A2=更改后第一组的密文, B1=修改前第二组的明文, B2=修改后第二组的明文, C=第二组的密文(需要保持不变)
因为：C=A1 ^ B1
因为：C=A2 ^ B2
而且：C应该保持不变
那么：A1 ^ B1 = A2 ^ B2
那么：A2 = A1 ^ B1 ^ B2

​ 然后把第一组的密文与第二组的密文合到一块发送个密码机，得到解密之后的内容，此时我们后面得到的明文必然已经是修改过的了(已经变成我们想要修改成为的内容了)，但是此时会发现前面的第一块明文解密出错了，这是因为第一块密文做了改变的原因。这时我们应该是已知第一块明文所需要成为的内容的，那就又变成了上面的相同的问题了，只需要修改初始向量IV的值即可，即A2 = A1 ^ B1 ^ B2，不过此时所求为目的IV。

A1=原来的iv, A2=目的iv, B1=修改前第一组的明文, B2=修改后第一组的明文, C=第一组的密文(需要保持不变)

CBC翻转攻击的条件

  1. 给出了初始iv的值，并且初始iv的值可以控制

  2. 要修改第二块明文的内容

  3. 有明文可求得一次密文

  4. 有密文可求得两次明文(且这两次解密的iv应是可控的)

是个简单的格密码，LWE问题，下面是题目的脚本

import random
import hashlib
from Crypto.Util.number import *
from Crypto.Cipher import AES
from secret import flag, V


def get_random_U(n):
    # U1为上三角行列式
    def get_U1():
        A = Matrix(ZZ, n, n)
        for i in range(0, n):
            for j in range(0, n):
                if i < j:
                    A[i, j] = random.randint(1, 1000)
                if i == j:
                    A[i, j] = 1
        return A
    # U2为下三角行列式
    def get_U2():
        A = Matrix(ZZ, n, n)
        for i in range(0, n):
            for j in range(0, n):
                if i > j:
                    A[i, j] = random.randint(1, 1000)
                if i == j:
                    A[i, j] = 1
        return A

    return get_U1() * get_U2()


def get_public_key():
    U = get_random_U(9)
    V = matrix(V)
    W = V * U
    return W


def get_random_r():
    n = 9
    delta = 4
    r = random_vector(ZZ, n, x=-delta + 1, y=delta)
    r = matrix(r)
    return r


def encrypt():
    M = [getPrime(10) for i in range(9)]
    m = matrix(M)
    W = get_public_key()
    r = get_random_r()
    e = m * W + r  # 有噪音的高斯消除问题(误差还原LWE问题)
    print("e =", e)
    print("W =", W)
    return M


def new_encrypt():
    M = encrypt()
    key = hashlib.sha256(str(M).encode()).digest()
    cipher = AES.new(key, AES.MODE_ECB)
    c = cipher.encrypt(flag).hex()
    print("c =", c)
